   function showproductid(value)
        {
            var id = value;
            $.ajax({
                type:'post',
                data: {item:id},
                contentType: 'json',
                success:function(){
                    swal("Seçilen Ürün ID:" + id, "Ürün Sepetinize Eklendi", "success");
                }
            });
        }
